import { ThemeCompositions } from '@teambit/documenter.theme.theme-compositions';
import { ErrorPage} from './error-page';

### Overview

An error page component for displaying error pages on bit.dev:

If the error code isn't supported, the alt "error image" is displayed

```js live
() => {
  return (
    <ThemeCompositions style={{ minHeight: 500, flexDirection: 'column' }}>
        <ErrorPage code={404} title="This is a 404 error page" />
    </ThemeCompositions>
  );
};
```
